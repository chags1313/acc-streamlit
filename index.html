let prevPoseLandmarks = null; let calorieCounter = 0; function onResults(results) { canvasCtx.save(); canvasCtx.clearRect(0, 0, canvasElement.width, canvasElement.height); if (results.poseLandmarks) { drawConnectors(canvasCtx, results.poseLandmarks, POSE_CONNECTIONS, {color: 'white'}); drawLandmarks(canvasCtx, results.poseLandmarks, { color: 'white', fillColor: 'rgb(255,255,255)', lineWidth: 2, radius: (landmark) => landmark.visibility < 0.1 ? 0 : 1, }); if (prevPoseLandmarks) { let activity = 0; const joints = [15, 17, 16, 18]; // indices for left and right wrists and elbows for (let joint of joints) { let dx = results.poseLandmarks[joint].x - prevPoseLandmarks[joint].x; let dy = results.poseLandmarks[joint].y - prevPoseLandmarks[joint].y; let dz = results.poseLandmarks[joint].z - prevPoseLandmarks[joint].z; activity += Math.sqrt(dx*dx + dy*dy + dz*dz); } calorieCounter += activity; } prevPoseLandmarks = results.poseLandmarks; calorieElement.textContent = Math.floor(calorieCounter * 10); // converting activity score to "calories" timeElement.textContent = ((Date.now() - startTime) / 60000).toFixed(2); } canvasCtx.restore(); }
